G = Graph(directed) # Creates a graph instance

G.add_node(id) # Creates a node with the given id for the graph G which we defined above.

G.add_edge(id_1, id_2) # Creates an edge between the nodes id_1 and id_2.
# If the graph is directed, then the edge is from id_1 to id_2.

G.remove_node(id) # Removes the node with the given id.

G.remove_edge(id_1, id_2) # Removes the edge between the nodes id_1 and id_2.
# If the graph is directed then the only the edge that is node id_1 to id_2 is removed.

G.set_node_attribute(node_id, attribute, value) # Sets the value of attribute for the node with the given id to the given value.
